SELECT
    GJE.SUBLEDGERVOUCHER,
	GJE.RECID,
	GJE.SUBLEDGERVOUCHERDATAAREAID,
	LJT1.NUMOFLINES,
	LJT1.JOURNALTYPE,
	LJT1.JOURNALTOTALCREDIT,
	LJT1.JOURNALTOTALDEBIT,
    COUNT(*) AS COUNT
FROM GENERALJOURNALENTRY GJE
JOIN LEDGERJOURNALTRANS LJT ON GJE.SUBLEDGERVOUCHER = LJT.VOUCHER AND LJT.DATAAREAID = GJE.SUBLEDGERVOUCHERDATAAREAID
